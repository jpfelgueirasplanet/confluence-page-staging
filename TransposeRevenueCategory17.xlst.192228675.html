Who should use this filter?

This filter should be used with any vendor who requires the actual nightly rate and not just the logis value.

What does it do?

The filter which sits within the Manifest for the vendor is responsible for making sure that the values in AmountBeforeTax and AmountAfterTax are correct according to the night rate value of the reservation.


<Base> Amount - Message Before the Transposition

<Rate EffectiveDate="2018-01-14" ExpireDate="2018-01-15" RateTimeUnit="Day" UnitMultiplier="1">
<Base AmountAfterTax="91.20" AmountBeforeTax="85.23" CurrencyCode="EUR"/>
<TPA_Extensions>
<px:RateAmountExtensions End="2018-01-15" Start="2018-01-14">
<px:RateAmount AmountAfterTax="96.00" AmountBeforeTax="89.26" RevenueCategoryCode="17"/>
<px:RateAmount AmountAfterTax="91.20" AmountBeforeTax="85.23" RevenueCategoryCode="9"/>
<px:RateAmount AmountAfterTax="4.80" AmountBeforeTax="4.03" RevenueCategoryCode="3"/>
</px:RateAmountExtensions>
</TPA_Extensions>
</Rate>

<Base> Amount - Message After the Transposition

<Rate EffectiveDate="2018-01-14" ExpireDate="2018-01-15" RateTimeUnit="Day" UnitMultiplier="1">
<Base AmountAfterTax="96.00" AmountBeforeTax="89.26" CurrencyCode="EUR"/>
<TPA_Extensions>
<px:RateAmountExtensions End="2018-01-15" Start="2018-01-14">
<px:RateAmount AmountAfterTax="96.00" AmountBeforeTax="89.26" RevenueCategoryCode="17"/>
<px:RateAmount AmountAfterTax="91.20" AmountBeforeTax="85.23" RevenueCategoryCode="9"/>
<px:RateAmount AmountAfterTax="4.80" AmountBeforeTax="4.03" RevenueCategoryCode="3"/>
</px:RateAmountExtensions>
</TPA_Extensions>
</Rate>

The revenue values from @RevenueCategoryCode=17 inside the <TPA_Extension> have been applied to @AmountBeforeTax and @AmountAfterTax in the <Base> section

Code behind the Filter

<?xml version="1.0" encoding="UTF-8" ?>
<xsl:transform xmlns:xsl="http://www.w3.org/1999/XSL/Transform" version="2.0"
xmlns:xs="http://www.w3.org/2001/XMLSchema"
xmlns:px="http://protel.io/htng/extensions"
exclude-result-prefixes="xs"
xmlns="http://www.opentravel.org/OTA/2003/05"
xpath-default-namespace="http://www.opentravel.org/OTA/2003/05">
<xsl:output encoding="UTF-8" indent="yes" method="xml" version="1.0"/>
<xsl:template match="@*|node()">
<xsl:copy>
<xsl:apply-templates select="@*|node()"/>
</xsl:copy>
</xsl:template>
<xsl:template match="RoomStay/Total">
<xsl:element name="{local-name()}" namespace="http://www.opentravel.org/OTA/2003/05">
<xsl:copy-of select="@*[name() != 'AmountBeforeTax' and name() != 'AmountAfterTax' and name() != 'AmountAfterTax']" />
<xsl:attribute name="AmountBeforeTax">
<xsl:value-of select="../TPA_Extensions/px:RateAmountExtensions/px:RateAmount[@RevenueCategoryCode = '17']/@AmountBeforeTax"/>
</xsl:attribute>
<xsl:attribute name="AmountAfterTax">
<xsl:value-of select="../TPA_Extensions/px:RateAmountExtensions/px:RateAmount[@RevenueCategoryCode = '17']/@AmountAfterTax"/>
</xsl:attribute>
<xsl:attribute name="CurrencyCode">
<xsl:value-of select="../TPA_Extensions/px:RateAmountExtensions/px:RateAmount[@RevenueCategoryCode = '17']/@CurrencyCode"/>
</xsl:attribute>
</xsl:element>
</xsl:template>
<xsl:template match="Rate">
<xsl:element name="{local-name()}" namespace="http://www.opentravel.org/OTA/2003/05">
<xsl:attribute name="EffectiveDate">
<xsl:value-of select="@EffectiveDate"/>
</xsl:attribute>
<xsl:attribute name="ExpireDate">
<xsl:value-of select="@ExpireDate"/>
</xsl:attribute>
<xsl:attribute name="RateTimeUnit">
<xsl:value-of select="@RateTimeUnit"/>
</xsl:attribute>
<xsl:attribute name="UnitMultiplier">
<xsl:value-of select="@UnitMultiplier"/>
</xsl:attribute>
<xsl:element name="Base" namespace="http://www.opentravel.org/OTA/2003/05">
<xsl:attribute name="AmountAfterTax">
<xsl:value-of select="TPA_Extensions/px:RateAmountExtensions/px:RateAmount[@RevenueCategoryCode = '17']/@AmountAfterTax" />
</xsl:attribute>
<xsl:attribute name="AmountBeforeTax">
<xsl:value-of select="TPA_Extensions/px:RateAmountExtensions/px:RateAmount[@RevenueCategoryCode = '17']/@AmountBeforeTax" />
</xsl:attribute>
<xsl:attribute name="CurrencyCode">
<xsl:value-of select="Base/@CurrencyCode" />
</xsl:attribute>
</xsl:element>
<xsl:element name="Total" namespace="http://www.opentravel.org/OTA/2003/05">
<xsl:attribute name="AmountAfterTax">
<xsl:value-of select="TPA_Extensions/px:RateAmountExtensions/px:RateAmount[@RevenueCategoryCode = '17']/@AmountAfterTax" />
</xsl:attribute>
<xsl:attribute name="AmountBeforeTax">
<xsl:value-of select="TPA_Extensions/px:RateAmountExtensions/px:RateAmount[@RevenueCategoryCode = '17']/@AmountBeforeTax" />
</xsl:attribute>
<xsl:attribute name="CurrencyCode">
<xsl:value-of select="Base/@CurrencyCode" />
</xsl:attribute>
</xsl:element>
<xsl:copy-of select="TPA_Extensions"/>
</xsl:element>
</xsl:template>
</xsl:transform>
