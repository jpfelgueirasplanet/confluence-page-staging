on prem | I/O Installation - Xsetup Validation

Once you have done the General Preparation, you can validate your xsetups by using this query.

In Microsoft SQL Studio

Use one of these depending on how the database schema is set up:

DB Schema dbo
BEGIN
IF((select count(*) from dbo.xsetup where mpehotel=0 AND xsection='pws' AND xkey='htng_request_url' AND xvalue='https://service.protel.io/ProtelApiService.ProtelApiServiceHttpsSoap12Endpoint') = 1)
PRINT '[Xsetup] GLOBAL - pws - htng_request_url --> OK';
ELSE
PRINT '[Xsetup] GLOBAL - pws - htng_request_url --> NOT OK';
IF((select count(*) from dbo.xsetup where mpehotel=0 AND xsection='pws' AND xkey='protelio_websocket_url' AND xvalue='wss://wsesb.protel.io') = 1)
PRINT '[Xsetup] GLOBAL - pws - protelio_websocket_url --> OK';
ELSE
PRINT '[Xsetup] GLOBAL - pws - protelio_websocket_url --> NOT OK';
IF((select count(*) from dbo.xsetup where mpehotel <> 0 AND xsection='pws' AND xkey='htng_logging_enabled') > 0)
PRINT '[Xsetup] LOCAL - pws - htng_logging_enabled --> OK';
ELSE
PRINT '[Xsetup] LOCAL - pws - htng_logging_enabled --> NOT OK';
IF((select count(*) from dbo.xsetup where mpehotel <> 0 AND xsection='pws' AND xkey='htng_tax_included') > 0)
PRINT '[Xsetup] LOCAL - pws - htng_tax_included --> OK';
ELSE
PRINT '[Xsetup] LOCAL - pws - htng_tax_included --> NOT OK';
IF((select count(*) from dbo.xsetup where mpehotel <> 0 AND xsection='invoice' AND xkey='allownegativesplitting' AND xvalue='1') > 0)
PRINT '[Xsetup] LOCAL - invoice - allownegativesplitting --> OK';
ELSE
PRINT '[Xsetup] LOCAL - invoice - allownegativesplitting --> NOT OK';
IF((select count(*) from dbo.xsetup where mpehotel=0 AND xsection='navigator' AND xkey='ratenewfixed' AND xvalue='1') = 1)
PRINT '[Xsetup] GLOBAL - navigator - ratenewfixed --> OK';
ELSE
PRINT '[Xsetup] GLOBAL - navigator - ratenewfixed --> NOT OK';
IF((select count(*) from dbo.xsetup where mpehotel <> 0 AND xsection='prforecast' AND xkey='inclallotment' AND xvalue='1') > 0)
PRINT '[Xsetup] LOCAL - prforecast - inclallotment --> OK';
ELSE
PRINT '[Xsetup] LOCAL - prforecast - inclallotment --> NOT OK';
IF((select count(*) from dbo.xsetup where mpehotel=0 AND xsection='prforecast' AND xkey='numberofdays' AND xvalue='5000') = 1)
PRINT '[Xsetup] GLOBAL - prforecast - numberofdays --> OK';
ELSE
PRINT '[Xsetup] GLOBAL - prforecast - numberofdays --> NOT OK';
IF((select count(*) from dbo.xsetup where mpehotel=0 AND xsection='system' AND xkey='ForbiddenRights' AND xvalue='079,397,069') = 1)
PRINT '[Xsetup] GLOBAL - system - ForbiddenRights --> OK';
ELSE
PRINT '[Xsetup] GLOBAL - system - ForbiddenRights --> NOT OK';
IF((select count(*) from dbo.pws_outboundmsg where sendlive <> 0) > 0)
PRINT '[IO Config] Live Mode OK';
ELSE
PRINT '[IO Config] Live Mode NOT OK';
END
